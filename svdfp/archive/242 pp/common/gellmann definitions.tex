% The 8 Gell-Mann matrices for SU(3)
\newcommand{ \gma }[0] {
\mat{rrr}{
    0 &    1 &    0 \\
    1 &    0 &    0 \\
 \ps0 & \ps0 & \ps0
} }

\newcommand{ \gmb }[0] {
\mat{rrr}{
 \ps0 & -i & \ps0 \\
    i &  0 &    0 \\
    0 &  0 &    0
} }

\newcommand{ \gmc }[0] {
\mat{rrr}{
    1 &  0 &    0 \\
 \ps0 & -1 & \ps0 \\
    0 &  0 &    0
} }

\newcommand{ \gmd }[0] {
\mat{rrr}{
 \ps0 & \ps0 & \ps1 \\
    0 &    0 &    0 \\
    1 &    0 &    0
} }

\newcommand{ \gme }[0] {
\mat{rrr}{
 \ps0 & \ps0 & -i \\
    0 &    0 &  0 \\
    i &    0 &  0
} }

\newcommand{ \gmf }[0] {
\mat{rrr}{
 \ps0 & \ps0 & \ps0 \\
    0 &    0 &    1 \\
    0 &    1 &    0
} }

\newcommand{ \gmg }[0] {
\mat{rrr}{
 \ps0 & \ps 0 & \ps0 \\
 \ps0 & \ps 0 &   -i \\
 \ps0 & \ps i & \ps0
} }

\newcommand{ \gmh }[0] {
\mat{rrr}{
    1 &    0 &  0 \\
    0 &    1 &  0 \\
 \ps0 & \ps0 & -2
} }
