\subsection{Hermitian matrices}
By definition
\begin{equation}
  \A{} = \A{*}
\end{equation}
If a matrix is Hermitian, it is also normal. Therefore
\begin{equation}
  \A{}\A{*} = \A{*}\A{}
\end{equation}
\begin{equation}
  \U{} \sig{2} \U{*} = \V{} \sig{2} \V{*}
\end{equation}
\begin{equation}
  \U{} = \V{}
\end{equation}
\begin{equation}
  \A{} = \U{} \sig{} \U{*}
\end{equation}
