% The 8 Gell-Mann matrices for SU(3)
\newcommand{ \gma } {
\left[
\begin{array}{ccc}
 0 & 1 & 0 \\
 1 & 0 & 0 \\
 0 & 0 & 0
\end{array}
\right]
}

\newcommand{ \gmb } {
\left[
\begin{array}{crc}
 0 & -i & 0 \\
 i & 0 & 0 \\
 0 & 0 & 0
\end{array}
\right]
}

\newcommand{ \gmc } {
\left[
\begin{array}{crc}
 1 & 0 & 0 \\
 0 & -1 & 0 \\
 0 & 0 & 0
\end{array}
\right]
}

\newcommand{ \gmd } {
\left[
\begin{array}{ccc}
 0 & 0 & 1 \\
 0 & 0 & 0 \\
 1 & 0 & 0
\end{array}
\right]
}
\newcommand{ \gme } {
\left[
\begin{array}{ccr}
 0 & 0 & -i \\
 0 & 0 & 0 \\
 i & 0 & 0
\end{array}
\right]
}
\newcommand{ \gmf } {
\left[
\begin{array}{ccc}
 0 & 0 & 0 \\
 0 & 0 & 1 \\
 0 & 1 & 0
\end{array}
\right]
}
\newcommand{ \gmg } {
\left[
\begin{array}{ccr}
 0 & 0 & 0 \\
 0 & 0 & -i \\
 0 & i & 0
\end{array}
\right]
}
\newcommand{ \gmh } {
\sthree
\left[
\begin{array}{ccr}
 1 & 0 & 0 \\
 0 & 1 & 0 \\
 0 & 0 & -2
\end{array}
\right]
}
