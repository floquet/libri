\subsubsection{$\by{2}{2}$ matrices}
\begin{equation}
  \begin{split}
  \A{} &= \mat{cc}{1&1\\0&1}\\
    \Y{} &= \frac{1}{\sqrt{10}}
    \left(
\begin{array}{cc}
 \sqrt{5+\sqrt{5}} & -\sqrt{5-\sqrt{5}} \\
 2 \sqrt{\frac{5}{5+\sqrt{5}}} & \sqrt{5+\sqrt{5}}
\end{array}
\right)\\
    \sig{}&=\frac{1}{2}
    \left(
\begin{array}{cc}
 1+\sqrt{5} & 0 \\
 0 & -1+\sqrt{5}
\end{array}
\right)\\
    \X{} &= \frac{1}{\sqrt{10}}
    \left(
\begin{array}{cc}
 \sqrt{5-\sqrt{5}} & -\sqrt{10 \left(\frac{1}{2}+\frac{1}{2 \sqrt{5}}\right)} \\
 \sqrt{5+\sqrt{5}} & 2 \sqrt{\frac{5}{5+\sqrt{5}}}
\end{array}
\right)
  \end{split}
\end{equation}
