\section{Arithmetic Cases}  %    S    S    S    S    S    S    S    S    S

  \begin{equation}   %  =   =   =   =   =
   \begin{split}
     y &= a_{1} x_{1} \pm a_{2} x_{2} \\
     \eps_{y}^{2} &=  a_{1}^{2} \eps_{1}^{2} + a_{2}^{2} \eps_{2}^{2}
   \end{split}
   %\label{eq:}
  \end{equation}

  \begin{equation}   %  =   =   =   =   =
   \begin{split}
     y &= a x_{1} x_{2} \\
     \eps_{y}^{2} &=  a^{2} \paren{ x_{1}^{2} \eps_{2}^{2} + x_{2}^{2} \eps_{1}^{2} }
   \end{split}
   %\label{eq:}
  \end{equation}

  \begin{equation}   %  =   =   =   =   =
   \begin{split}
     y &= a \frac{x_{1}} {x_{2}} \\
     \eps_{y}^{2} &=  a^{2} \paren{ \frac{\eps_{1}^{2}} {x_{2}^{2}} + \frac{\eps_{2}^{2}} {x_{1}^{2}} }
   \end{split}
   %\label{eq:}
  \end{equation}
