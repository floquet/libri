\newcommand{ \matrixaXt}[0]
{
    \rstwo
    \mat{ rr }

\newcommand{ \matrixaXf}[0]
{
    \recip{\sqrt{2}}
    \mat{ r >{\columncolor{ltgray}}r }
      {
       \rstwo &  \rstwo \\
      -\rstwo &  \phantom{-}\rstwo
      }
}

\newcommand{ \matrixaY}[0]
{
    \mat{ r >{\columncolor{ltgray}}r >{\columncolor{ltgray}}r }
      {
       \rsthree & 0                 &  \frac{2}{ \sqrt{6} } \\[5pt]
      -\rsthree & \rstwo            &  \rssix \\[5pt]
       \rsthree & \phantom{-}\rstwo & -\rssix
      }
}

\newcommand{ \matrixaYt}[0]
{
    \rst
    \mat{ rrr }
      {
       \rsthree           & -\rsthree &  \rsthree } \\[5pt]
       \rowcolor{ltgray}
       \phantom{-}0       &  \rstwo   &  \rssix     \\[5pt]
       \rowcolor{ltgray}
       \frac{2}{ \sqrt{6} &  \rstwo   & -\rssix
      }
}
